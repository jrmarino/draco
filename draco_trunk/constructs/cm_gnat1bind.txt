############## generate_einfo_h ##############
# This function generates the dependency einfo.h

function (generate_einfo_h)

   file (COPY ${DIR_ADA}/einfo.ads ${DIR_ADA}/einfo.adb ${DIR_ADA}/xeinfo.adb
         DESTINATION ${CMAKE_CURRENT_BINARY_DIR}/einfo)

   add_custom_command (
      OUTPUT einfo.h
      COMMAND ${CMAKE_COMMAND} -E chdir einfo ${pass_compiler} -c xeinfo.adb
      COMMAND ${CMAKE_COMMAND} -E chdir einfo ${pass_binder} -x xeinfo.ali
      COMMAND ${CMAKE_COMMAND} -E chdir einfo ${pass_linker} xeinfo.ali
      COMMAND ${CMAKE_COMMAND} -E chdir einfo ./xeinfo ../einfo.h
      DEPENDS einfo/xeinfo.adb einfo/einfo.ads einfo/xeinfo.adb
   )

endfunction (generate_einfo_h)

function (generate_nmake_sb)

   file (COPY  ${DIR_ADA}/xutil.ads
               ${DIR_ADA}/xutil.adb
               ${DIR_ADA}/sinfo.ads
               ${DIR_ADA}/nmake.adt
               ${DIR_ADA}/xnmake.adb
         DESTINATION ${CMAKE_CURRENT_BINARY_DIR}/nmake)

   add_custom_command (
      OUTPUT nmake.ads nmake.adb
      COMMAND ${CMAKE_COMMAND} -E chdir nmake ${pass_compiler} -c xutil.adb
      COMMAND ${CMAKE_COMMAND} -E chdir nmake ${pass_compiler} -c xnmake.adb
      COMMAND ${CMAKE_COMMAND} -E chdir nmake ${pass_binder} -x xnmake.ali
      COMMAND ${CMAKE_COMMAND} -E chdir nmake ${pass_linker} xnmake.ali
      COMMAND ${CMAKE_COMMAND} -E chdir nmake ./xnmake -b ./nmake.adb
      COMMAND ${CMAKE_COMMAND} -E chdir nmake ./xnmake -s ../nmake.ads
      COMMAND ${CMAKE_COMMAND} -E copy  nmake/nmake.adb .
      DEPENDS
               nmake/xutil.ads
               nmake/xutil.adb
               nmake/sinfo.ads
               nmake/nmake.adt
               nmake/xnmake.adb
   )

endfunction (generate_nmake_sb)




############## generate_sdefault ##############
# This function generates the dependency sdefault.adb

function (generate_sdefault)

   set (SF sdefault.adb)

   file (WRITE  ${SF} "pragma Style_Checks (Off);\n")
   file (APPEND ${SF} "with Osint; use Osint;\n")
   file (APPEND ${SF} "package body Sdefault is\n")
   file (APPEND ${SF} "   S0 : constant String := \"${draco_prefix}/\";\n")
   file (APPEND ${SF} "   S1 : constant String := \"${draco_prefix}/lib/draco-${DRACO_VERSION}/adainclude/\";\n")
   file (APPEND ${SF} "   S2 : constant String := \"${draco_prefix}/lib/draco-${DRACO_VERSION}/adalib/\";\n")
   file (APPEND ${SF} "   S3 : constant String := \"${TARGET_TRIPLET}\";\n")
   file (APPEND ${SF} "   S4 : constant String := \"${draco_prefix}/lib/draco-${DRACO_VERSION}/\";\n")
   file (APPEND ${SF} "   function Include_Dir_Default_Name return String_Ptr is\n")
   file (APPEND ${SF} "   begin\n")
   file (APPEND ${SF} "      return Relocate_Path (S0, S1);\n")
   file (APPEND ${SF} "   end Include_Dir_Default_Name;\n")
   file (APPEND ${SF} "   function Object_Dir_Default_Name return String_Ptr is\n")
   file (APPEND ${SF} "   begin\n")
   file (APPEND ${SF} "      return Relocate_Path (S0, S2);\n")
   file (APPEND ${SF} "   end Object_Dir_Default_Name;\n")
   file (APPEND ${SF} "   function Target_Name return String_Ptr is\n")
   file (APPEND ${SF} "   begin\n")
   file (APPEND ${SF} "      return new String'(S3);\n")
   file (APPEND ${SF} "   end Target_Name;\n")
   file (APPEND ${SF} "   function Search_Dir_Prefix return String_Ptr is\n")
   file (APPEND ${SF} "   begin\n")
   file (APPEND ${SF} "      return Relocate_Path (S0, S4);\n")
   file (APPEND ${SF} "   end Search_Dir_Prefix;\n")
   file (APPEND ${SF} "end Sdefault;\n")

endfunction (generate_sdefault)
